def func_count(text):
    words = re.findall("[a-zA-Z_]+", text)
    return len([word for word in words if len(word)>=2])
In [180]:
def func_unique(text):
    words = re.findall("[a-zA-Z_]+", text)
    unique = set(words)
    return len([uniq for uniq in unique if len(uniq)>=2])

def func_characters(text):
    full_text = list(text)
    return len([full for full in full_text if full != ' '])
In [189]:
def func_punc(text):
    punc = 0
    for i in text:
        if i in string.punctuation:
            punc += 1
    return punc
In [192]:
def func_digit(text):
    digit = 0
    for i in text:
        if i.isdigit():
            digit+=1
    return digit
In [197]:
def func_special(text):
    hashtag = 0
    monkey = 0
    dolar = 0
    
    for i in text:
        if i == '@':
            monkey+= 1
        elif i == '#':
            hashtag+= 1
        elif i == '$':
            dolar+=1
    return hashtag, monkey, dolar
